# 정수형 자료형과 실수형 자료형

```c++
#include <iostream>
#include <climits>

using namespace std;

int main() {

    //정수형 : 소수부가 없는 수
    // 음의 정수, 0, 양의 정수

    int n_int = INT_MAX;
    short n_shrot = SHRT_MAX;
    long n_long = LONG_MAX;
    long long n_llong = LLONG_MAX;

    cout << "int는 " << sizeof n_int << "바이트 입니다." << endl;
    cout << "이 바이트의 최대값은 " << n_int << "이다." << endl;

    cout << "short는 " << sizeof n_shrot << "바이트 입니다." << endl;
    cout << "이 바이트의 최대값은 " << n_shrot << "이다." << endl;

    cout << "long는 " << sizeof n_long << "바이트 입니다." << endl;
    cout << "이 바이트의 최대값은 " << n_long << "이다." << endl;

    cout << "long long는 " << sizeof n_llong << "바이트 입니다." << endl;
    cout << "이 바이트의 최대값은 " << n_llong << "이다." << endl;

    return 0;
}
```
